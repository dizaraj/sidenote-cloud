Database Setup Required

You must run this SQL snippet in your Supabase SQL Editor to create the correct table for SideNote Cloud:

CREATE TABLE public.notes (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  local_id text UNIQUE NOT NULL,
  title text,
  body text,
  urls jsonb,
  created_at timestamp with time zone DEFAULT now()
);

-- Enable RLS and setup Anon access if needed (optional for simple usage)
ALTER TABLE public.notes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Enable all for anon" ON public.notes FOR ALL USING (true) WITH CHECK (true);

-- Add is_archived Flag to Notes
ALTER TABLE public.notes ADD COLUMN is_archived boolean DEFAULT false;
